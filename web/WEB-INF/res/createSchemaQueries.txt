CREATE TABLE `tasks` (
  `tasks_id` int(11) NOT NULL AUTO_INCREMENT,

  `tasks_date` date NOT NULL,

  `user_id` int(11) NOT NULL,
  PRIMARY KEY (`tasks_id`),
